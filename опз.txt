using System;

namespace MyName
{
    class Program
    {
        static void Main()
        {
            string str = Console.ReadLine();
            List<object> operations = new List<object>();
            Stack<string> stack = new Stack<string>();
            foreach (var symbol in str)
            {
                if (char.IsDigit(symbol))
                {
                    operations.Add(symbol);
                }
                else
                {
                    if (!(stack.Contains("*")) || !(stack.Contains("/")))
                    {
                        stack.Push(symbol.ToString());
                    }
                    else
                    {
                        while (stack.Count!=0)
                        {
                            string operand = stack.Pop();
                            operations.Add(operand);
                        }
                        stack.Push(symbol.ToString());
                    }
                }
            }

            foreach (var variable in operations)
            {
                Console.Write(variable);
            }
            foreach (var elem in stack)
            {
                Console.Write(elem);
            }

        }
    }
}
